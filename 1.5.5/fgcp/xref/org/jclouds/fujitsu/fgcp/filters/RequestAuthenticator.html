62  /**
63   * Generates and signs the access key id and adds the mandatory http header and
64   * request parameters to the request.
65   * 
66   * @author Dies Koper
67   */
68  @Singleton
69  public class RequestAuthenticator implements HttpRequestFilter, RequestSigner {
70  
71     @Resource
72     @Named(Constants.LOGGER_SIGNATURE)
73     private Logger signatureLog = Logger.NULL;
74  
75     final Provider<Calendar> calendarProvider;
76     final Signature signer;
77     final String apiVersion;
78  
79     public String signatureVersion = "1.0";
80     public String signatureMethod = "SHA1withRSA";
81  
82     private HttpUtils utils;
83  
84     @Inject
85     public RequestAuthenticator(@TimeStamp Provider<Calendar> calendarProvider, Provider<KeyStore> keyStoreProvider,
86           @Credential String keyPassword, HttpUtils utils, SignatureWire signatureWire, @ApiVersion String apiVersion)
87           throws NoSuchAlgorithmException, InvalidKeyException, KeyStoreException, UnrecoverableKeyException {
88        this.calendarProvider = checkNotNull(calendarProvider);
89        this.utils = checkNotNull(utils, "utils");
90        this.apiVersion = checkNotNull(apiVersion, "apiVersion");
91  
92        signer = Signature.getInstance(signatureMethod);
93  
94        KeyStore keyStore = checkNotNull(keyStoreProvider).get();
95        String alias = keyStore.aliases().nextElement(); // there should be only one private key
96        PrivateKey privateKey = (PrivateKey) keyStore.getKey(alias,
97              keyPassword.toCharArray());
98  
99        signer.initSign(privateKey);
100    }

102    public HttpRequest filter(HttpRequest request) throws HttpException {
103       checkNotNull(request, "request must be present");
104 
105       utils.logRequest(signatureLog, request, ">>");
106 
107       // create accesskeyid
108       String accessKeyId = createStringToSign(request);
109       String signature = sign(accessKeyId);
110 
111       // only "en" and "ja" are allowed
112       String lang = Locale.JAPANESE.getLanguage().equals(
113             Locale.getDefault().getLanguage()) ? Locale.JAPANESE
114             .getLanguage() : Locale.ENGLISH.getLanguage();
115 
116       if (HttpMethod.GET.equals(request.getMethod())) {
117          Multimap<String, String> decodedParams = queryParser().apply(request.getEndpoint().getRawQuery());
118 
119          if (!decodedParams.containsKey(RequestParameters.VERSION)) {
120             decodedParams.put(RequestParameters.VERSION, apiVersion);
121          }
122          decodedParams.put(RequestParameters.LOCALE, lang);
123          decodedParams.put(RequestParameters.ACCESS_KEY_ID, accessKeyId);
124          decodedParams.put(RequestParameters.SIGNATURE, signature);
125          request = request.toBuilder().replaceQueryParams(decodedParams).build();
126       } else {
127 
128          String payload = request.getPayload().getRawContent().toString();
129          payload = createXmlElementWithValue(payload,
130                RequestParameters.VERSION, apiVersion);
131          payload = createXmlElementWithValue(payload,
132                RequestParameters.LOCALE, lang);
133          payload = createXmlElementWithValue(payload,
134                RequestParameters.ACCESS_KEY_ID, accessKeyId);
135          payload = createXmlElementWithValue(payload,
136                RequestParameters.SIGNATURE, signature);
137 
138          // ensure there are no other query params left
139          request.setPayload(payload);
140          request.getPayload().getContentMetadata()
141                .setContentType(MediaType.TEXT_XML);
142       }
143 
144       // may need to do this elsewhere (see ConvertToGaeRequest)
145       HttpRequest filteredRequest = request.toBuilder()
146             .replaceHeader(HttpHeaders.USER_AGENT, "OViSS-API-CLIENT")
147             .build();
148 
149       utils.logRequest(signatureLog, filteredRequest, ">>->");
150 
151       return filteredRequest;
152    }

154    String createXmlElementWithValue(String payload, String tag, String value) {
155       String startTag = String.format("<%s>", tag);
156       String endTag = String.format("</%s>", tag);
157 
158       return payload.replace(startTag + endTag, startTag + value + endTag);
159    }

169    @VisibleForTesting
170    public String sign(String stringToSign) {
171       String signed;
172 
173       try {
174          signer.update(stringToSign.getBytes(Charsets.UTF_8));
175          signed = CryptoStreams.base64(signer.sign());
176       } catch (SignatureException e) {
177          throw new HttpException("error signing request", e);
178       }
179       // if (signatureWire.enabled())
180       // signatureWire.input(Strings2.toInputStream(signed));
181 
182       return signed;
183    }

185    @VisibleForTesting
186    public String generateAccessKeyId() {
187       Calendar cal = calendarProvider.get();
188       String timezone = cal.getTimeZone().getDisplayName(Locale.ENGLISH);
189       String expires = String.valueOf(cal.getTime().getTime());
190 
191       String signatureData = String.format("%s&%s&%s&%s", timezone, expires,
192             signatureVersion, signatureMethod);
193       String accessKeyId = CryptoStreams.base64(signatureData.getBytes(Charsets.UTF_8));
194       return accessKeyId.replace("\n", "\r\n");
195    }
196 
197    @Override
198    public String createStringToSign(HttpRequest input) {
199       return generateAccessKeyId();
200    }
